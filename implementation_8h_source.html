1682// base64_options are used to specify the base64 encoding options.
1683using base64_options = uint64_t;
1684enum : base64_options {
1685 base64_default = 0, /* standard base64 format (with padding) */
1686 base64_url = 1, /* base64url format (no padding) */
1687 base64_reverse_padding = 2, /* modifier for base64_default and base64_url */
1688 base64_default_no_padding =
1689 base64_default |
1690 base64_reverse_padding, /* standard base64 format without padding */
1691 base64_url_with_padding =
1692 base64_url | base64_reverse_padding, /* base64url with padding */
1693};
1694
1695// last_chunk_handling_options are used to specify the handling of the last
1696// chunk in base64 decoding.
1697// https://tc39.es/proposal-arraybuffer-base64/spec/#sec-frombase64
1698enum last_chunk_handling_options : uint64_t {
1699 loose = 0, /* standard base64 format, decode partial final chunk */
1700 strict =
1701 1, /* error when the last chunk is partial, 2 or 3 chars, and unpadded */
1702 stop_before_partial =
1703 2, /* if the last chunk is partial (2 or 3 chars), ignore it (no error) */
1704};
